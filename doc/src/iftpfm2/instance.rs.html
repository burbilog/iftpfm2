16// Signal the existing process to terminate gracefully
17fn signal_process_to_terminate(socket_path: &str, grace_seconds: u64) -> io::Result<()> {
18    // Use lsof to find process using the socket
19    let output = Command::new("lsof")
20        .arg("-t")  // Output only PID
21        .arg(socket_path)
22        .output()?;
23
24    if !output.status.success() {
25        return Err(io::Error::new(
26            io::ErrorKind::Other,
27            "Failed to find process using lsof"
28        ));
29    }
30
31    let pid_str = String::from_utf8_lossy(&output.stdout).trim().to_string();
32    if pid_str.is_empty() {
33        return Err(io::Error::new(
34            io::ErrorKind::NotFound,
35            "No process found using the socket"
36        ));
37    }
38
39    let _ = log(&format!("Found old instance with PID {}, sending termination signal", pid_str));
40
41    // Set the shutdown flag for our own process if we're signaling ourselves
42    // This case should ideally not happen if check_single_instance is called correctly,
43    // but it's a safeguard.
44    let our_pid = std::process::id().to_string();
45    if pid_str == our_pid {
46        request_shutdown();
47        return Ok(());
48    }
49
50    // Send SIGTERM to allow graceful shutdown
51    let term_output = Command::new("kill")
52        .arg("-15")  // SIGTERM for graceful termination
53        .arg(&pid_str)
54        .output()?;
55
56    if !term_output.status.success() {
57        let stderr = String::from_utf8_lossy(&term_output.stderr);
58        return Err(io::Error::new(
59            io::ErrorKind::Other,
60            format!("Failed to send termination signal to process {}: {}", pid_str, stderr)
61        ));
62    }
63
64    let _ = log(&format!("Successfully sent termination signal to old instance with PID {}", pid_str));
65
66    // Wait for up to grace_seconds for the process to terminate
67    for i in 1..=(grace_seconds * 2) { // Check twice per second
68        std::thread::sleep(std::time::Duration::from_millis(500));
69
70        // Check if the process is still running
71        let check_output = Command::new("kill")
72            .arg("-0")  // Check if process exists
73            .arg(&pid_str)
74            .output()?;
75
76        if !check_output.status.success() {
77            let _ = log(&format!("Old instance with PID {} has terminated gracefully", pid_str));
78            return Ok(());
79        }
80
81        if i % 2 == 0 { // Log every second
82            let _ = log(&format!("Waiting for old instance with PID {} to terminate ({} of {} seconds)...",
83                pid_str, i/2, grace_seconds));
84        }
85    }
86
87    // If process didn't terminate after timeout, use SIGKILL as last resort
88    let _ = log(&format!("Old instance with PID {} did not terminate gracefully, forcing termination", pid_str));
89    let kill_output = Command::new("kill")
90        .arg("-9")  // SIGKILL for forced termination
91        .arg(&pid_str)
92        .output()?;
93
94    if !kill_output.status.success() {
95        let stderr = String::from_utf8_lossy(&kill_output.stderr);
96        return Err(io::Error::new(
97            io::ErrorKind::Other,
98            format!("Failed to force termination of process {}: {}", pid_str, stderr)
99        ));
100    }
101
102    let _ = log(&format!("Forcibly terminated old instance with PID {}", pid_str));
103    std::thread::sleep(std::time::Duration::from_millis(500)); // Give OS a moment
104
105    Ok(())
106}

108/// Ensures only one instance runs at a time
109///
110/// # Behavior
111/// - Creates lock file with PID
112/// - Listens on Unix socket for shutdown requests
113/// - Handles cleanup on exit
114///
115/// # Errors
116/// - If socket creation fails
117/// - If PID file can't be written
118///
119/// # Panics
120/// If signal handler registration fails
121pub fn check_single_instance(grace_seconds: u64) -> io::Result<()> {
122    let socket_path = format!("/tmp/{}.sock", crate::PROGRAM_NAME); // Using PROGRAM_NAME from lib.rs
123
124    // Try to connect to existing socket
125    if UnixStream::connect(&socket_path).is_ok() {
126        let _ = log(&format!("Another instance is running, new instance PID {} requesting graceful termination of old one.",
127            std::process::id()));
128
129        // Try to signal the process to terminate gracefully
130        if let Err(e) = signal_process_to_terminate(&socket_path, grace_seconds) {
131            let _ = log(&format!("Failed to signal old process: {}. Stale socket/pid files might exist.", e));
132            // Even if signaling fails, we might be able to remove the socket if it's stale.
133        }
134
135        // Attempt to clean up the socket file after signaling (or if signaling failed but socket was stale)
136        // This is important so the new instance can bind to it.
137        let _ = std::fs::remove_file(&socket_path);
138        let _ = log(&format!("Removed old socket file: {}", socket_path));
139
140    } else {
141        // If connection failed, it might be because the socket file is stale (no one listening).
142        // Clean it up before trying to bind.
143        let _ = std::fs::remove_file(&socket_path);
144    }
145
146    // Create a new socket file for this instance
147    let listener = UnixListener::bind(&socket_path)?;
148    let _ = log(&format!("Created new socket file: {}", socket_path));
149
150    // Write our PID to a common PID file location
151    let pid_path = format!("/tmp/{}.pid", crate::PROGRAM_NAME); // Using PROGRAM_NAME from lib.rs
152    let mut pid_file = File::create(&pid_path)?;
153    pid_file.write_all(std::process::id().to_string().as_bytes())?;
154    let _ = log(&format!("Written current PID {} to {}", std::process::id(), pid_path));
155
156    // Set up signal handler for SIGINT (Ctrl+C) and SIGTERM
157    // NOTE: This handler is async-signal-safe. It only sets atomic flags.
158    // Logging is deferred to the main thread.
159    ctrlc::set_handler(|| {
160        // Signal type 1 = SIGINT (Ctrl+C)
161        crate::shutdown::request_shutdown_with_signal(1);
162    }).expect("Error setting signal handler");
163
164    // Spawn a thread to listen on the socket for shutdown commands from new instances.
165    let handle = thread::spawn(move || {
166        for stream in listener.incoming() {
167            match stream {
168                Ok(mut stream) => {
169                    let mut buffer = [0; 8]; // Expect "SHUTDOWN"
170                    if let Ok(size) = stream.read(&mut buffer) {
171                        if size == 8 && &buffer[..] == b"SHUTDOWN" {
172                            let _ = log(&format!("Received 'SHUTDOWN' command on socket. PID {} initiating self-shutdown.",
173                                std::process::id()));
174                            request_shutdown();
175                            break; // Exit listener thread
176                        }
177                    }
178                }
179                Err(e) => {
180                    let _ = log(&format!("Error accepting incoming connection on socket: {}", e));
181                    // Depending on the error, might want to break or continue.
182                    // For now, continue to try accepting more connections.
183                }
184            }
185        }
186        let _ = log("Socket listener thread exiting.");
187    });
188
189    // Store the handle for later cleanup
190    if let Ok(mut guard) = LISTENER_HANDLE.lock() {
191        *guard = Some(handle);
192    }
193
194    Ok(())
195}

197/// Joins the socket listener thread
198///
199/// Attempts to wait for the listener thread to finish.
200/// The thread may be blocked on listener.incoming(), so we only wait briefly.
201/// If the thread is still blocked, we continue with cleanup anyway - the OS
202/// will terminate the thread when the process exits.
203/// Should be called during cleanup.
204pub fn join_listener_thread() {
205    if let Ok(mut handle_guard) = LISTENER_HANDLE.lock() {
206        if let Some(handle) = handle_guard.take() {
207            // Try to join in a separate thread that we timeout
208            std::thread::spawn(move || {
209                let _ = handle.join();
210            });
211            // Don't wait - the listener thread is often blocked on incoming()
212            // The OS will clean it up when the process exits
213        }
214    }
215}

217/// Cleans up single instance lock files
218///
219/// Removes:
220/// - Unix domain socket (/tmp/{PROGRAM_NAME}.sock)
221/// - PID file (/tmp/{PROGRAM_NAME}.pid)
222///
223/// Called automatically on program exit (e.g., via scopeguard in main)
224pub fn cleanup_lock_file() {
225    let socket_path = format!("/tmp/{}.sock", crate::PROGRAM_NAME); // Using PROGRAM_NAME from lib.rs
226    let pid_path = format!("/tmp/{}.pid", crate::PROGRAM_NAME); // Using PROGRAM_NAME from lib.rs
227
228    let _ = log(&format!("Cleaning up lock files: {} and {}", socket_path, pid_path));
229
230    if let Err(e) = std::fs::remove_file(&socket_path) {
231        if e.kind() != std::io::ErrorKind::NotFound {
232            let _ = log(&format!("Failed to remove socket file {}: {}", socket_path, e));
233        }
234    }
235    if let Err(e) = std::fs::remove_file(&pid_path) {
236        if e.kind() != std::io::ErrorKind::NotFound {
237            let _ = log(&format!("Failed to remove pid file {}: {}", pid_path, e));
238        }
239    }
240}
