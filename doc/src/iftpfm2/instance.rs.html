10// Signal the existing process to terminate gracefully
11fn signal_process_to_terminate(socket_path: &str, grace_seconds: u64) -> io::Result<()> {
12    // Use lsof to find process using the socket
13    let output = Command::new("lsof")
14        .arg("-t")  // Output only PID
15        .arg(socket_path)
16        .output()?;
17
18    if !output.status.success() {
19        return Err(io::Error::new(
20            io::ErrorKind::Other,
21            "Failed to find process using lsof"
22        ));
23    }
24
25    let pid_str = String::from_utf8_lossy(&output.stdout).trim().to_string();
26    if pid_str.is_empty() {
27        return Err(io::Error::new(
28            io::ErrorKind::NotFound,
29            "No process found using the socket"
30        ));
31    }
32
33    log(&format!("Found old instance with PID {}, sending termination signal", pid_str)).unwrap();
34
35    // Set the shutdown flag for our own process if we're signaling ourselves
36    // This case should ideally not happen if check_single_instance is called correctly,
37    // but it's a safeguard.
38    let our_pid = std::process::id().to_string();
39    if pid_str == our_pid {
40        request_shutdown();
41        return Ok(());
42    }
43
44    // Send SIGTERM to allow graceful shutdown
45    let term_output = Command::new("kill")
46        .arg("-15")  // SIGTERM for graceful termination
47        .arg(&pid_str)
48        .output()?;
49
50    if !term_output.status.success() {
51        let stderr = String::from_utf8_lossy(&term_output.stderr);
52        return Err(io::Error::new(
53            io::ErrorKind::Other,
54            format!("Failed to send termination signal to process {}: {}", pid_str, stderr)
55        ));
56    }
57
58    log(&format!("Successfully sent termination signal to old instance with PID {}", pid_str)).unwrap();
59
60    // Wait for up to grace_seconds for the process to terminate
61    for i in 1..=(grace_seconds * 2) { // Check twice per second
62        std::thread::sleep(std::time::Duration::from_millis(500));
63
64        // Check if the process is still running
65        let check_output = Command::new("kill")
66            .arg("-0")  // Check if process exists
67            .arg(&pid_str)
68            .output()?;
69
70        if !check_output.status.success() {
71            log(&format!("Old instance with PID {} has terminated gracefully", pid_str)).unwrap();
72            return Ok(());
73        }
74
75        if i % 2 == 0 { // Log every second
76            log(&format!("Waiting for old instance with PID {} to terminate ({} of {} seconds)...",
77                pid_str, i/2, grace_seconds)).unwrap();
78        }
79    }
80
81    // If process didn't terminate after timeout, use SIGKILL as last resort
82    log(&format!("Old instance with PID {} did not terminate gracefully, forcing termination", pid_str)).unwrap();
83    let kill_output = Command::new("kill")
84        .arg("-9")  // SIGKILL for forced termination
85        .arg(&pid_str)
86        .output()?;
87
88    if !kill_output.status.success() {
89        let stderr = String::from_utf8_lossy(&kill_output.stderr);
90        return Err(io::Error::new(
91            io::ErrorKind::Other,
92            format!("Failed to force termination of process {}: {}", pid_str, stderr)
93        ));
94    }
95
96    log(&format!("Forcibly terminated old instance with PID {}", pid_str)).unwrap();
97    std::thread::sleep(std::time::Duration::from_millis(500)); // Give OS a moment
98
99    Ok(())
100}

102/// Ensures only one instance runs at a time
103///
104/// # Behavior
105/// - Creates lock file with PID
106/// - Listens on Unix socket for shutdown requests
107/// - Handles cleanup on exit
108///
109/// # Errors
110/// - If socket creation fails
111/// - If PID file can't be written
112///
113/// # Panics
114/// If signal handler registration fails
115pub fn check_single_instance(grace_seconds: u64) -> io::Result<()> {
116    let socket_path = format!("/tmp/{}.sock", crate::PROGRAM_NAME); // Using PROGRAM_NAME from lib.rs
117
118    // Try to connect to existing socket
119    if UnixStream::connect(&socket_path).is_ok() {
120        log(&format!("Another instance is running, new instance PID {} requesting graceful termination of old one.",
121            std::process::id())).unwrap();
122
123        // Try to signal the process to terminate gracefully
124        if let Err(e) = signal_process_to_terminate(&socket_path, grace_seconds) {
125            log(&format!("Failed to signal old process: {}. Stale socket/pid files might exist.", e)).unwrap();
126            // Even if signaling fails, we might be able to remove the socket if it's stale.
127        }
128
129        // Attempt to clean up the socket file after signaling (or if signaling failed but socket was stale)
130        // This is important so the new instance can bind to it.
131        let _ = std::fs::remove_file(&socket_path);
132        log(&format!("Removed old socket file: {}", socket_path)).unwrap();
133
134    } else {
135        // If connection failed, it might be because the socket file is stale (no one listening).
136        // Clean it up before trying to bind.
137        let _ = std::fs::remove_file(&socket_path);
138    }
139
140    // Create a new socket file for this instance
141    let listener = UnixListener::bind(&socket_path)?;
142    log(&format!("Created new socket file: {}", socket_path)).unwrap();
143
144    // Write our PID to a common PID file location
145    let pid_path = format!("/tmp/{}.pid", crate::PROGRAM_NAME); // Using PROGRAM_NAME from lib.rs
146    let mut pid_file = File::create(&pid_path)?;
147    pid_file.write_all(std::process::id().to_string().as_bytes())?;
148    log(&format!("Written current PID {} to {}", std::process::id(), pid_path)).unwrap();
149
150    // Set up signal handler for SIGINT (Ctrl+C) and SIGTERM
151    let current_pid = std::process::id();
152    ctrlc::set_handler(move || {
153        log(&format!("Received termination signal (Ctrl+C or SIGTERM), PID {} shutting down gracefully", current_pid)).unwrap();
154        request_shutdown();
155        // The cleanup_lock_file will be called via scopeguard in main.
156        // Consider if additional cleanup is needed here or if it's robust enough.
157    }).expect("Error setting signal handler");
158
159    // Spawn a thread to listen on the socket for shutdown commands from new instances.
160    std::thread::spawn(move || {
161        for stream in listener.incoming() {
162            match stream {
163                Ok(mut stream) => {
164                    let mut buffer = [0; 8]; // Expect "SHUTDOWN"
165                    if let Ok(size) = stream.read(&mut buffer) {
166                        if size == 8 && &buffer[..] == b"SHUTDOWN" {
167                            log(&format!("Received 'SHUTDOWN' command on socket. PID {} initiating self-shutdown.",
168                                std::process::id())).unwrap();
169                            request_shutdown();
170                            break; // Exit listener thread
171                        }
172                    }
173                }
174                Err(e) => {
175                    log(&format!("Error accepting incoming connection on socket: {}", e)).unwrap();
176                    // Depending on the error, might want to break or continue.
177                    // For now, continue to try accepting more connections.
178                }
179            }
180        }
181        log("Socket listener thread exiting.").unwrap();
182    });
183
184    Ok(())
185}

187/// Cleans up single instance lock files
188///
189/// Removes:
190/// - Unix domain socket (/tmp/{PROGRAM_NAME}.sock)
191/// - PID file (/tmp/{PROGRAM_NAME}.pid)
192///
193/// Called automatically on program exit (e.g., via scopeguard in main)
194pub fn cleanup_lock_file() {
195    let socket_path = format!("/tmp/{}.sock", crate::PROGRAM_NAME); // Using PROGRAM_NAME from lib.rs
196    let pid_path = format!("/tmp/{}.pid", crate::PROGRAM_NAME); // Using PROGRAM_NAME from lib.rs
197
198    log(&format!("Cleaning up lock files: {} and {}", socket_path, pid_path)).unwrap();
199
200    if let Err(e) = std::fs::remove_file(&socket_path) {
201        if e.kind() != std::io::ErrorKind::NotFound {
202            log(&format!("Failed to remove socket file {}: {}", socket_path, e)).unwrap();
203        }
204    }
205    if let Err(e) = std::fs::remove_file(&pid_path) {
206        if e.kind() != std::io::ErrorKind::NotFound {
207            log(&format!("Failed to remove pid file {}: {}", pid_path, e)).unwrap();
208        }
209    }
210}
