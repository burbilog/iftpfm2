9// LOG_FILE is a thread-safe, lazily initialized global variable
10// It holds an Option<String> representing the path to the log file (if set)
11// The Mutex ensures thread-safe access to this value
12/// Global log file path protected by Mutex
13///
14/// Thread-safe storage for optional log file path.
15/// When None, logs go to stdout.
16pub static LOG_FILE: Lazy<Mutex<Option<String>>> = Lazy::new(|| Mutex::new(None));
17
18/// Global cached file handle protected by Mutex
19///
20/// Thread-safe storage for optional buffered writer to log file.
21/// When None, either no log file is set or we haven't opened it yet.
22static LOG_FILE_HANDLE: Lazy<Mutex<Option<BufWriter<File>>>> = Lazy::new(|| Mutex::new(None));

24/// Global debug mode flag (AtomicBool for lock-free reads)
25///
26/// When true, debug messages are logged. When false, log_debug() is a no-op.
27/// This allows debug logging to be enabled/disabled at runtime without performance impact.
28pub static DEBUG_MODE: AtomicBool = AtomicBool::new(false);
29
30/// Enable or disable debug mode
31///
32/// # Arguments
33/// * `enabled` - true to enable debug logging, false to disable
34pub fn set_debug_mode(enabled: bool) {
35    DEBUG_MODE.store(enabled, Ordering::SeqCst);
36}
37
38/// Check if debug mode is enabled
39///
40/// # Returns
41/// * `bool` - true if debug logging is enabled
42pub fn is_debug_enabled() -> bool {
43    DEBUG_MODE.load(Ordering::SeqCst)
44}

46/// Logs a message to either a file or stdout
47///
48/// This function takes a message as input and logs it with a timestamp.
49/// If a log file has been set (using set_log_file), the message is appended to that file.
50/// Otherwise, the message is printed to stdout.
51///
52/// # Arguments
53///
54/// * `message` - The message to be logged
55///
56/// # Returns
57///
58/// * `io::Result<()>` - Ok if the logging was successful, Err otherwise
59///   Logs a message.
60///
61/// Logs a message with timestamp to configured output
62///
63/// # Arguments
64/// * `message` - The message to log
65///
66/// # Returns
67/// * `io::Result<()>` - Ok on success, Err if writing fails
68///
69/// # Example
70/// ```text
71/// // log("Starting transfer").unwrap();
72/// ```
73pub fn log(message: &str) -> io::Result<()> {
74    log_with_thread(message, None)
75}

77/// Logs a message with timestamp and optional thread ID
78///
79/// Used when running in parallel mode to distinguish threads
80///
81/// # Arguments
82/// * `message` - The message to log (accepts anything that can be referenced as a str)
83/// * `thread_id` - Optional thread identifier
84///
85/// # Returns
86/// * `io::Result<()>` - Ok on success, Err if writing fails
87///
88/// # Example
89/// ```text
90/// // log_with_thread("Thread started", Some(1)).unwrap();
91/// // log_with_thread(format!("Value: {}", x), Some(1)).unwrap();
92/// ```
93pub fn log_with_thread<T: AsRef<str>>(message: T, thread_id: Option<usize>) -> io::Result<()> {
94    // Generate a timestamp for the log message
95    let timestamp = Local::now().format("%Y-%m-%d %H:%M:%S").to_string();
96    let message_ref = message.as_ref();
97    let log_message = match thread_id {
98        Some(tid) => format!("{} [T{}] {}\n", timestamp, tid, message_ref),
99        None => format!("{} {}\n", timestamp, message_ref),
100    };
101
102    // Lock the mutex and check if a log file has been set
103    // Handle poisoned mutex by recovering or using a fallback
104    let log_file_result = LOG_FILE.lock();
105    let log_file_guard = match log_file_result {
106        Ok(guard) => guard,
107        Err(poisoned) => {
108            // Recover from poisoned mutex, taking the value
109            poisoned.into_inner()
110        }
111    };
112
113    // Clone the log_file path so we can drop the guard before locking LOG_FILE_HANDLE
114    let log_file_clone = log_file_guard.as_ref().cloned();
115    drop(log_file_guard);
116
117    if let Some(log_file) = log_file_clone {
118        // Lock the file handle mutex, handling poisoning
119        let handle_result = LOG_FILE_HANDLE.lock();
120        let mut handle_guard: std::sync::MutexGuard<'_, Option<BufWriter<File>>> = match handle_result {
121            Ok(guard) => guard,
122            Err(poisoned) => poisoned.into_inner(),
123        };
124
125        // If handle is not yet opened or was closed, open it
126        if handle_guard.is_none() {
127            let file = OpenOptions::new()
128                .create(true)
129                .append(true)
130                .open(&log_file)?;
131            *handle_guard = Some(BufWriter::new(file));
132        }
133
134        // Write to the cached handle, with fallback to stderr on failure
135        let write_result = if let Some(ref mut writer) = *handle_guard {
136            writer.write_all(log_message.as_bytes()).and_then(|_| writer.flush())
137        } else {
138            Ok(())
139        };
140
141        if let Err(e) = write_result {
142            // Fallback to stderr if file logging fails
143            eprintln!("[LOGGING FAILED: {}] {}", log_file, e);
144            eprintln!("{}", log_message.trim_end());
145        }
146    } else {
147        // If no log file is set, print the message to stdout.
148        // The original code used println!() with a message already ending in \n,
149        // resulting in a double newline. Restoring that behavior.
150        println!("{}", log_message);
151    }
152
153    Ok(())
154}

156/// Logs a debug message (only when debug mode is enabled)
157///
158/// This function is a no-op when debug mode is disabled, avoiding unnecessary
159/// string formatting and I/O. Use this for verbose diagnostic information.
160///
161/// # Arguments
162/// * `message` - The message to log (accepts anything that can be referenced as a str)
163/// * `thread_id` - Optional thread identifier
164///
165/// # Returns
166/// * `io::Result<()>` - Ok on success, Err if writing fails (always Ok when disabled)
167///
168/// # Example
169/// ```text
170/// // log_debug("Temp file path: /tmp/xxx", None);
171/// // set_debug_mode(true);  // Enable debug mode first
172/// // log_debug(format!("Size: {} bytes", size), Some(1));
173/// ```
174pub fn log_debug<T: AsRef<str>>(message: T, thread_id: Option<usize>) -> io::Result<()> {
175    if !is_debug_enabled() {
176        return Ok(());
177    }
178    log_with_thread(message, thread_id)
179}

181/// Sets the path for the log file
182///
183/// This function updates the global LOG_FILE variable with the provided path.
184/// Subsequent calls to the log function will write to this file.
185///
186/// # Arguments
187///
188/// * `path` - A path-like object representing the location of the log file
189pub fn set_log_file<P: AsRef<Path>>(path: P) {
190    // Convert the path to a string and update the LOG_FILE
191    let path_str = path.as_ref().to_str().expect("Path is not valid UTF-8");
192
193    // Update the log file path, handling poisoned mutex
194    let result = LOG_FILE.lock();
195    let mut guard = match result {
196        Ok(g) => g,
197        Err(poisoned) => poisoned.into_inner(),
198    };
199    *guard = Some(path_str.to_string());
200    drop(guard);
201
202    // Clear any cached file handle since the path has changed
203    let result = LOG_FILE_HANDLE.lock();
204    let mut handle_guard: std::sync::MutexGuard<'_, Option<BufWriter<File>>> = match result {
205        Ok(g) => g,
206        Err(poisoned) => poisoned.into_inner(),
207    };
208    *handle_guard = None;
209}
