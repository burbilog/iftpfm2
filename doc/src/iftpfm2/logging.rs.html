8// LOG_FILE is a thread-safe, lazily initialized global variable
9// It holds an Option<String> representing the path to the log file (if set)
10// The Mutex ensures thread-safe access to this value
11/// Global log file path protected by Mutex
12///
13/// Thread-safe storage for optional log file path.
14/// When None, logs go to stdout.
15pub static LOG_FILE: Lazy<Mutex<Option<String>>> = Lazy::new(|| Mutex::new(None));

17/// Logs a message to either a file or stdout
18///
19/// This function takes a message as input and logs it with a timestamp.
20/// If a log file has been set (using set_log_file), the message is appended to that file.
21/// Otherwise, the message is printed to stdout.
22///
23/// # Arguments
24///
25/// * `message` - The message to be logged
26///
27/// # Returns
28///
29/// * `io::Result<()>` - Ok if the logging was successful, Err otherwise
30/// Logs a message with timestamp to configured output
31///
32/// # Arguments
33/// * `message` - The message to log
34///
35/// # Returns
36/// * `io::Result<()>` - Ok on success, Err if writing fails
37///
38/// # Example
39/// ```text
40/// // log("Starting transfer").unwrap();
41/// ```
42pub fn log(message: &str) -> io::Result<()> {
43    log_with_thread(message, None)
44}
45
46/// Logs a message with timestamp and optional thread ID
47///
48/// Used when running in parallel mode to distinguish threads
49///
50/// # Arguments
51/// * `message` - The message to log
52/// * `thread_id` - Optional thread identifier
53///
54/// # Returns
55/// * `io::Result<()>` - Ok on success, Err if writing fails
56///
57/// # Example
58/// ```text
59/// // log_with_thread("Thread started", Some(1)).unwrap();
60/// ```
61pub fn log_with_thread(message: &str, thread_id: Option<usize>) -> io::Result<()> {
62    // Generate a timestamp for the log message
63    let timestamp = Local::now().format("%Y-%m-%d %H:%M:%S").to_string();
64    let log_message = match thread_id {
65        Some(tid) => format!("{} [T{}] {}\n", timestamp, tid, message),
66        None => format!("{} {}\n", timestamp, message),
67    };
68
69    // Lock the mutex and check if a log file has been set
70    match &*LOG_FILE.lock().unwrap() {
71        Some(log_file) => {
72            // If a log file is set, append the message to the file
73            let mut file = OpenOptions::new()
74                .create(true)
75                .append(true)
76                .open(log_file)?;
77            file.write_all(log_message.as_bytes())?;
78        }
79        None => {
80            // If no log file is set, print the message to stdout.
81            // The original code used println!() with a message already ending in \n,
82            // resulting in a double newline. Restoring that behavior.
83            println!("{}", log_message);
84        }
85    }
86
87    Ok(())
88}

90/// Sets the path for the log file
91///
92/// This function updates the global LOG_FILE variable with the provided path.
93/// Subsequent calls to the log function will write to this file.
94///
95/// # Arguments
96///
97/// * `path` - A path-like object representing the location of the log file
98pub fn set_log_file<P: AsRef<Path>>(path: P) {
99    // Convert the path to a string and update the LOG_FILE
100    let path_str = path.as_ref().to_str().expect("Path is not valid UTF-8");
101    *LOG_FILE.lock().unwrap() = Some(path_str.to_string());
102}
